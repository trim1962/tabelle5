\newcommand{\CDpm}[1]{
\ifnum #1>0
+#1\else
#1
\fi
}
\newcommand{\xunodueE}[3]{
\FPset\ca{#1}
\FPset\cb{#2}
\FPset\cc{#3}
\FPneg\menob\cb
\FPround\menob{\menob}{0}
\FPeval\bquad{abs(\cb) * abs(\cb)}
\FPround\bquad{\bquad}{0}
%\FPeval\qac{4*(\ca*\cc)}
\FPadd\duea\ca\ca%
\FPmul\qac\duea\cc 
\FPadd\qac\qac\qac%
\FPneg\mqac\qac
\FPround\mqac{\mqac}{0}
\FPround\duea{\duea}{0}
\dfrac{\menob\pm\sqrt{\bquad\CDpm{\mqac}}}{\duea}}
\newcommand{\xunodueS}[3]{
\FPset\ca{#1}
\FPset\cb{#2}
\FPset\cc{#3}
\FPqsolve\xone\xtwo\ca\cb\cc
\FPeval\xone{clip(round(xone:4))}%
\FPeval\xtwo{clip(round(xtwo:4))}%
x_1=\xone\quad x_2=\xtwo
}
\newcommand{\xunodueDeltaE}[3]{
	\FPset\ca{#1}
	\FPset\cb{#2}
	\FPset\cc{#3}
	%b^2
	\FPmul\bquad\cb\cb
	%2a
	\FPadd\duea\ca\ca
	%2ac
	\FPmul\qac\duea\cc
	%4ac
	\FPadd\qac\qac\qac%
	%-4ac
	\FPneg\mqac\qac
	%b^2-4ac
	\FPadd\deltas\bquad\mqac
	\FPround\bquad{\bquad}{0}
	\FPround\mqac{\mqac}{0}
	\FPround\deltas{\deltas}{0}
	\Delta=b^2-4ac=\bquad\CDpm{\mqac}=\deltas
}
\newcommand{\xunodueDeltaS}[3]{
	\FPset\ca{#1}
	\FPset\cb{#2}
	\FPset\cc{#3}
	%b^2
	\FPmul\bquad\cb\cb
	%2a
	\FPadd\duea\ca\ca
	%2ac
	\FPmul\qac\duea\cc
	%4ac
	\FPadd\qac\qac\qac%
	%-4ac
	\FPneg\mqac\qac
	%b^2-4ac
	\FPadd\deltas\bquad\mqac
	\FPround\deltas{\deltas}{0}
	\Delta=\deltas
}

\newcommand{\xunodueEE}[3]{
	\FPset\ca{#1}
	\FPset\cb{#2}
	\FPset\cc{#3}
	%-b
	\FPneg\menob\cb
	%b^2
	\FPmul\bquad\cb\cb
%2a
	\FPadd\duea\ca\ca%
%2ac
	\FPmul\qac\duea\cc 
%4ac
	\FPadd\qac\qac\qac%
%-4ac
	\FPneg\mqac\qac
%b^2-4ac
\FPadd\deltas\bquad\mqac
	\FPround\bquad{\bquad}{0}
	\FPround\menob{\menob}{0}
	\FPround\deltas{\deltas}{0}
	\FPround\duea{\duea}{0}
	\dfrac{\menob\pm\sqrt{\deltas}}{\duea}}
