\clip(-10.,-15.) rectangle (10.,15.);

200,domain=-10.0:-1.1] plot(\x,{(2.0*(\x)^(2.0)+(\x)+1.0)/((\x)+1.0)});

200,domain=-0.99:10.0] plot(\x,{(2.0*(\x)^(2.0)+(\x)+1.0)/((\x)+1.0)});
